represent the image mask as the parameters of a mixture-of-histograms generative model¶

#K = number of objects
K = len(np.unique(mask_array))

##to create prior, count the number of pixels in each document
L = [sum(sum(mask_array==ii)).astype('double') for ii in np.arange(K)+1]
print 'total number of pixels in each document:'
print L
print

prior_dist = np.array(L/V).reshape((1,K)) ##fraction of pixels in each object
print 'object prior:' 
print prior_dist

##to create the likelihood distributions, treat each object mask as the parameters of a multinomial

##this just reformats the mask the way we want it to be
def format_mask_as_lkhd():
    return np.hstack([np.atleast_2d(np.array(mask_array==ii).flatten()).T for ii in np.arange(K)+1])

##divide by total number of pixels to get a true probability instead of a count
lkhd_dist = format_mask_as_lkhd()/L

##If any pixel has 0 likelihood, set it to a very tiny number. this avoids lots of irritating bugs.
lkhd_dist = lkhd_dist + np.finfo(np.double).tiny

##re-normalize to make sure each column still sums to 1
lkhd_dist = lkhd_dist/lkhd_dist.sum(axis=0, keepdims=True)

sample multiple "probes" from the "true " model¶

In [72]: 
probes = np.zeros((V,M))

##use prior to determine number of probes per object
probes_per_object = np.random.multinomial(M, np.squeeze(prior_dist))

##generate object patches
cnt = 0
for ii,nn in enumerate(probes_per_object):
    for jj in range(nn):
        dx = np.random.choice(np.arange(V), size=P, replace=False, p=np.atleast_1d(lkhd_dist[:,ii]))
        probes[dx, cnt] = 1
        cnt += 1

##check a few probes
figure()
plain_plot(gca())
title('first probe')
imshow(probes[:,0].reshape(N))
figure()
plain_plot(gca())
title('last probe')
imshow(probes[:,M-1].reshape(N))
